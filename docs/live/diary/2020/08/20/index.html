    3. 使用 go mod

      go mod init github.com/ikingye/test

    4. 使用 cobra，搭建一个框架

      cobra init --pkg-name github.com/ikingye/test

    5. 创建 Makefile

      PROJECTNAME=$(shell basename "$(PWD)")

## dev: Start in development mode.
dev:
  go run main.go

## staging: Start in staging mode.
staging:
  go run main.go

.PHONY: help
all: help
help: Makefile
  @echo
  @echo "Choose a command run in "$(PROJECTNAME)":"
  @echo
  @sed -n 's/^##//p' $< | column -t -s ':' |  sed -e 's/^/ /'
  @echo
