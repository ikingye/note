    类型转换 # 

    package main

import (
    "strconv"
)

func main() {
    // string 转 int32
    num, err := strconv.ParseInt(`123456`, 10, 32)
    if err != nil {
        panic(err)
    }
    println(num)

    // int32 转 string
    str := strconv.FormatInt(123456, 10)
    println(str)
}
