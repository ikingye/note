    零值 #

    在 Go 语言中， 布尔类型的零值（初始值）为 false， 数值类型的零值为 0， 字符串类型的零值为空字符串 “"， 而指针、切片、映射、通道、函数和接口的零值则是 nil。

    循环变量 goroutines 错误 # 

    // 错误
for _, val := range values {
	go func() {
		fmt.Println(val)
	}()
}

// 使用局部变量
for i := range valslice {
	val := valslice[i]
	go func() {
		fmt.Println(val)
	}()
}

// 使用参数传递
for _, val := range values {
	go func(val interface{}) {
		fmt.Println(val)
	}(val)
}


func (v *val) MyMethod() {
        fmt.Println(v)
}

// 错误
for _, val := range values {
	go val.MyMethod()
}

// 正确（使用局部变量）
for _, val := range values {
    newVal := val
	go newVal.MyMethod()
}
